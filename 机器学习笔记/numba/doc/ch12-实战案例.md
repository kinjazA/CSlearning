## 12.1 机器学习：特征工程加速

### 12.1.1 案例 1：距离矩阵计算

**场景**：KNN、聚类等算法需要计算样本之间的距离矩阵。

```python
import numpy as np
from numba import njit, prange

# 纯 NumPy 版本
def distance_matrix_numpy(X):
    """X: (n_samples, n_features)"""
    n = X.shape[0]
    D = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            diff = X[i] - X[j]
            D[i, j] = np.sqrt(np.sum(diff * diff))
            D[j, i] = D[i, j]
    return D

# Numba 版本（串行）
@njit
def distance_matrix_numba(X):
    n, d = X.shape
    D = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            dist = 0.0
            for k in range(d):
                diff = X[i, k] - X[j, k]
                dist += diff * diff
            dist = np.sqrt(dist)
            D[i, j] = dist
            D[j, i] = dist
    return D

# Numba 并行版本
@njit(parallel=True)
def distance_matrix_parallel(X):
    n, d = X.shape
    D = np.zeros((n, n))
    for i in prange(n):
        for j in range(i+1, n):
            dist = 0.0
            for k in range(d):
                diff = X[i, k] - X[j, k]
                dist += diff * diff
            dist = np.sqrt(dist)
            D[i, j] = dist
            D[j, i] = dist
    return D

# 性能对比
X = np.random.randn(1000, 50).astype(np.float64)

# NumPy: ~几秒
# Numba 串行: ~0.1秒 (快 20-50×)
# Numba 并行: ~0.02秒 (快 100-200×)
```

### 12.1.2 案例 2：滑动窗口特征

**场景**：时序数据的滑动窗口统计（均值、标准差、最大最小值）。

```python
@njit
def rolling_features(data, window):
    """
    对每个时间点，计算前 window 个数据的统计特征
    返回: (rolling_mean, rolling_std, rolling_max, rolling_min)
    """
    n = len(data)
    means = np.empty(n)
    stds = np.empty(n)
    maxs = np.empty(n)
    mins = np.empty(n)
    
    for i in range(n):
        start = max(0, i - window + 1)
        window_data = data[start:i+1]
        
        means[i] = np.mean(window_data)
        stds[i] = np.std(window_data)
        maxs[i] = np.max(window_data)
        mins[i] = np.min(window_data)
    
    return means, stds, maxs, mins

# 使用
data = np.random.randn(100000)
mean, std, max_val, min_val = rolling_features(data, window=50)
```

### 12.1.3 案例 3：自定义距离度量

**场景**：实现非标准距离度量（如 DTW、Mahalanobis）。

```python
@njit
def dtw_distance(s1, s2):
    """动态时间规整距离"""
    n, m = len(s1), len(s2)
    dtw = np.full((n+1, m+1), np.inf)
    dtw[0, 0] = 0.0
    
    for i in range(1, n+1):
        for j in range(1, m+1):
            cost = abs(s1[i-1] - s2[j-1])
            dtw[i, j] = cost + min(dtw[i-1, j],    # 插入
                                   dtw[i, j-1],    # 删除
                                   dtw[i-1, j-1])  # 匹配
    
    return dtw[n, m]

# 使用
s1 = np.array([1.0, 2.0, 3.0, 4.0])
s2 = np.array([1.0, 2.2, 3.1, 4.5, 5.0])
dist = dtw_distance(s1, s2)
```

### 12.1.4 案例 4：K-Means 聚类核心

**场景**：加速 K-Means 的分配步骤。

```python
@njit(parallel=True)
def assign_clusters(X, centroids):
    """
    为每个样本分配最近的簇
    X: (n_samples, n_features)
    centroids: (n_clusters, n_features)
    返回: labels (n_samples,)
    """
    n_samples = X.shape[0]
    n_clusters = centroids.shape[0]
    n_features = X.shape[1]
    labels = np.empty(n_samples, dtype=np.int32)
    
    for i in prange(n_samples):
        min_dist = np.inf
        best_cluster = 0
        
        for k in range(n_clusters):
            dist = 0.0
            for j in range(n_features):
                diff = X[i, j] - centroids[k, j]
                dist += diff * diff
            
            if dist < min_dist:
                min_dist = dist
                best_cluster = k
        
        labels[i] = best_cluster
    
    return labels

@njit
def update_centroids(X, labels, n_clusters):
    """更新簇中心"""
    n_samples, n_features = X.shape
    centroids = np.zeros((n_clusters, n_features))
    counts = np.zeros(n_clusters)
    
    for i in range(n_samples):
        label = labels[i]
        counts[label] += 1
        for j in range(n_features):
            centroids[label, j] += X[i, j]
    
    for k in range(n_clusters):
        if counts[k] > 0:
            for j in range(n_features):
                centroids[k, j] /= counts[k]
    
    return centroids
```

## 12.2 蒙特卡罗模拟

### 12.2.1 案例 5：期权定价（Black-Scholes Monte Carlo）

**场景**：用蒙特卡罗模拟对欧式期权定价。

```python
@njit
def european_call_monte_carlo(S0, K, T, r, sigma, n_simulations):
    """
    欧式看涨期权 Monte Carlo 定价
    S0: 当前股价
    K: 行权价
    T: 到期时间
    r: 无风险利率
    sigma: 波动率
    n_simulations: 模拟次数
    """
    payoffs = np.empty(n_simulations)
    
    for i in range(n_simulations):
        # 生成最终股价（几何布朗运动）
        Z = np.random.randn()
        ST = S0 * np.exp((r - 0.5 * sigma**2) * T + sigma * np.sqrt(T) * Z)
        
        # 期权到期收益
        payoffs[i] = max(ST - K, 0.0)
    
    # 折现到当前
    option_price = np.exp(-r * T) * np.mean(payoffs)
    return option_price

# 并行版本
@njit(parallel=True)
def european_call_monte_carlo_parallel(S0, K, T, r, sigma, n_simulations):
    payoffs = np.empty(n_simulations)
    
    for i in prange(n_simulations):
        Z = np.random.randn()
        ST = S0 * np.exp((r - 0.5 * sigma**2) * T + sigma * np.sqrt(T) * Z)
        payoffs[i] = max(ST - K, 0.0)
    
    option_price = np.exp(-r * T) * np.mean(payoffs)
    return option_price

# 使用
price = european_call_monte_carlo_parallel(
    S0=100,      # 当前股价
    K=105,       # 行权价
    T=1.0,       # 1年到期
    r=0.05,      # 5% 无风险利率
    sigma=0.2,   # 20% 波动率
    n_simulations=1_000_000
)
print(f"期权价格: {price:.4f}")
```

### 12.2.2 案例 6：π 值估算

**场景**：用 Monte Carlo 方法估算圆周率。

```python
@njit(parallel=True)
def estimate_pi_monte_carlo(n_samples):
    """
    用单位圆内点的比例估算 π
    π/4 ≈ (落在圆内的点数) / (总点数)
    """
    count_inside = 0
    
    for i in prange(n_samples):
        x = np.random.rand()
        y = np.random.rand()
        
        if x*x + y*y <= 1.0:
            count_inside += 1
    
    return 4.0 * count_inside / n_samples

# 使用
pi_estimate = estimate_pi_monte_carlo(10_000_000)
print(f"π 估算值: {pi_estimate:.6f}")
print(f"真实 π:   {np.pi:.6f}")
print(f"误差:     {abs(pi_estimate - np.pi):.6f}")
```

@njit
def random_walk_2d(n_steps, n_particles):
    """
    二维随机游走
    返回: (n_particles, n_steps, 2) 轨迹数组
    """
    trajectories = np.zeros((n_particles, n_steps, 2))
    
    for p in range(n_particles):
        x, y = 0.0, 0.0
        for t in range(n_steps):
            # 随机方向
            angle = np.random.rand() * 2 * np.pi
            step_size = 0.1
            
            x += step_size * np.cos(angle)
            y += step_size * np.sin(angle)
            
            trajectories[p, t, 0] = x
            trajectories[p, t, 1] = y
    
    return trajectories

# 使用

## 12.3 数值计算

### 12.3.1 案例 8：数值微分

**场景**：用有限差分法计算梯度。

```python
@njit
def numerical_gradient(f, x, eps=1e-5):
    """
    计算函数 f 在点 x 的梯度
    f: 标量函数
    x: numpy 数组
    """
    n = len(x)
    grad = np.empty(n)
    
    for i in range(n):
        x_plus = x.copy()
        x_minus = x.copy()
        
        x_plus[i] += eps
        x_minus[i] -= eps
        
        grad[i] = (f(x_plus) - f(x_minus)) / (2 * eps)
    
    return grad

# 示例：计算 f(x) = sum(x^2) 的梯度
@njit
def quadratic(x):
    return np.sum(x * x)

x = np.array([1.0, 2.0, 3.0])
grad = numerical_gradient(quadratic, x)
# 真实梯度: [2, 4, 6]
```

### 12.3.2 案例 9：梯度下降优化

**场景**：实现梯度下降算法。

```python
@njit
def gradient_descent(f, grad_f, x0, lr=0.01, max_iter=1000, tol=1e-6):
    """
    梯度下降优化
    f: 目标函数
    grad_f: 梯度函数
    x0: 初始点
    lr: 学习率
    max_iter: 最大迭代次数
    tol: 收敛容差
    """
    x = x0.copy()
    history = np.empty((max_iter, len(x0)))
    
    for iter in range(max_iter):
        grad = grad_f(x)
        x_new = x - lr * grad
        
        history[iter] = x_new
        
        # 检查收敛
        if np.linalg.norm(x_new - x) < tol:
            return x_new, history[:iter+1]
        
        x = x_new
    
    return x, history

# 示例：最小化 f(x) = (x-3)^2 + (y-2)^2
@njit
def rosenbrock(x):
    return (x[0] - 3)**2 + (x[1] - 2)**2

@njit
def rosenbrock_grad(x):
    return np.array([2*(x[0] - 3), 2*(x[1] - 2)])

x0 = np.array([0.0, 0.0])
x_opt, history = gradient_descent(rosenbrock, rosenbrock_grad, x0)
print(f"最优解: {x_opt}")  # 应该接近 [3, 2]
```

### 12.3.3 案例 10：常微分方程求解（Euler 方法）

**场景**：求解 dy/dt = f(t, y)。

```python
@njit
def euler_method(f, y0, t_span, n_steps):
    """
    Euler 方法求解 ODE
    f: dy/dt = f(t, y)
    y0: 初始值
    t_span: (t_start, t_end)
    n_steps: 时间步数
    """
    t_start, t_end = t_span
    dt = (t_end - t_start) / n_steps
    
    t = np.linspace(t_start, t_end, n_steps + 1)
    y = np.empty(n_steps + 1)
    y[0] = y0
    
    for i in range(n_steps):
        y[i+1] = y[i] + dt * f(t[i], y[i])
    
    return t, y

# 示例：求解 dy/dt = -y, y(0) = 1
# 真实解: y = e^(-t)
@njit
def exponential_decay(t, y):
    return -y

t, y = euler_method(exponential_decay, y0=1.0, t_span=(0, 5), n_steps=1000)
# 对比真实解：np.exp(-t)
```

### 12.3.4 案例 11：数值积分（梯形法则）

**场景**：计算定积分 ∫f(x)dx。

```python
@njit
def trapezoidal_rule(f, a, b, n):
    """
    梯形法则计算积分
    f: 被积函数
    a, b: 积分区间
    n: 分割数
    """
    h = (b - a) / n
    integral = 0.5 * (f(a) + f(b))
    
    for i in range(1, n):
        x = a + i * h
        integral += f(x)
    
    integral *= h
    return integral

# 示例：计算 ∫(0 to π) sin(x) dx = 2
@njit
def sine(x):
    return np.sin(x)

result = trapezoidal_rule(sine, 0, np.pi, n=1000)
print(f"积分结果: {result:.6f}")  # 应该接近 2
```

## 12.4 深入案例：粒子滤波（Particle Filter）

### 12.4.1 粒子滤波基础理论

**核心思想**：用一组带权重的"粒子"（样本）来表示状态的概率分布，通过粒子的演化来估计隐藏状态。

```
经典滤波问题：
  已知：观测序列 z₁, z₂, ..., zₜ（有噪声）
  求：真实状态 x₁, x₂, ..., xₜ

传统方法：
  - 卡尔曼滤波：假设线性 + 高斯噪声
  - 扩展卡尔曼：线性化非线性系统

粒子滤波：
  - 不需要线性假设
  - 不需要高斯假设
  - 用蒙特卡罗采样表示任意分布
```

**算法框架**（Sequential Importance Resampling, SIR）：

```
初始化：N 个粒子 {x₀⁽ⁱ⁾, w₀⁽ⁱ⁾}，i=1...N

for t = 1, 2, ..., T:
    1. 预测 (Prediction)：
       x_t⁽ⁱ⁾ ~ p(x_t | x_{t-1}⁽ⁱ⁾)
       根据状态转移模型预测新位置

    2. 更新 (Update)：
       w_t⁽ⁱ⁾ ∝ w_{t-1}⁽ⁱ⁾ · p(z_t | x_t⁽ⁱ⁾)
       根据观测似然更新权重

    3. 归一化 (Normalize)：
       w_t⁽ⁱ⁾ = w_t⁽ⁱ⁾ / Σⱼ w_t⁽ʲ⁾

    4. 估计 (Estimate)：
       x̂_t = Σᵢ w_t⁽ⁱ⁾ · x_t⁽ⁱ⁾

    5. 重采样 (Resampling)：
       当有效粒子数 N_eff < N_threshold 时
       从当前粒子按权重重新采样
```

### 12.4.2 粒子滤波的变体

| 变体 | 特点 | 适用场景 |
|------|------|---------|
| **SIS (Sequential Importance Sampling)** | 无重采样，权重退化严重 | 仅理论研究 |
| **SIR (Sequential Importance Resampling)** | 每步都重采样，经典方法 | 通用 |
| **Auxiliary PF** | 提前用观测信息选择粒子 | 状态变化剧烈 |
| **Regularized PF** | 重采样后加核平滑 | 防止粒子贫化 |
| **Rao-Blackwellized PF** | 部分状态解析求解 | 线性子结构系统 |
| **Unscented PF** | 结合 UKF，用 sigma 点采样 | 高维非线性 |

### 12.4.3 重采样策略对比

| 策略 | 复杂度 | 特点 |
|------|--------|------|
| **Multinomial** | O(N log N) | 简单，但方差大 |
| **Systematic** | O(N) | 低方差，推荐 |
| **Residual** | O(N) | 保留高权重粒子 |
| **Stratified** | O(N) | 分层采样，方差最小 |

### 12.4.4 多 Seed 似然加权粒子滤波

**问题背景**（对应你在 ch05 提到的场景）：

在机器学习建模中，粒子滤波器生成的随机特征需要可复现——但 Numba 的线程级 RNG 导致多线程调度不确定，即使设了全局 seed，每次运行结果也不同（±0.1 RMSE 抖动）。

**解决方案**：多 Seed 似然加权策略

```
思路：
  1. 用 K 个不同的 seed 运行 K 次粒子滤波
  2. 每次运行产生一个状态估计 x̂ₖ
  3. 用观测似然对 K 个估计加权平均
  4. 最终输出：加权平均估计 + 置信区间

优势：
  - 每个 seed 内部确定性（方案 1，ch05 §5.6.5）
  - K 次运行取平均降方差（方案 2，ch05 §5.6.6）
  - 似然加权让"好"的估计贡献更大
```

## 12.5 完整实现：多 Seed 似然加权粒子滤波

### 12.5.1 状态空间模型定义

假设一个简单的一维跟踪问题：

```
状态方程：x_t = x_{t-1} + v_t，v_t ~ N(0, σ_process²)
观测方程：z_t = x_t + w_t，w_t ~ N(0, σ_measure²)

例：跟踪一个沿直线匀速运动的物体
  - x_t：物体真实位置
  - z_t：传感器观测位置（有噪声）
```

### 12.5.2 核心函数：单次粒子滤波

```python
from numba import njit
import numpy as np

@njit
def systematic_resample(weights):
    """
    系统重采样（低方差）
    返回：N 个索引，表示重采样后的粒子来源
    """
    N = len(weights)
    positions = (np.arange(N) + np.random.rand()) / N
    
    cumulative_sum = np.cumsum(weights)
    indices = np.zeros(N, dtype=np.int32)
    
    i, j = 0, 0
    while i < N:
        if positions[i] < cumulative_sum[j]:
            indices[i] = j
            i += 1
        else:
            j += 1
    
    return indices


@njit
def effective_sample_size(weights):
    """计算有效粒子数"""
    return 1.0 / np.sum(weights ** 2)


@njit
def particle_filter_single_seed(
    measurements,       # 观测序列 (T,)
    n_particles,        # 粒子数
    process_noise,      # 过程噪声标准差 σ_process
    measurement_noise,  # 观测噪声标准差 σ_measure
    initial_state,      # 初始真实状态（用于初始化粒子分布）
    seed                # 随机种子
):
    """
    单次确定性粒子滤波
    
    返回:
        estimates: 状态估计序列 (T,)
        particles_history: 粒子历史 (T, n_particles) 用于可视化
        weights_history: 权重历史 (T, n_particles)
    """
    np.random.seed(seed)  # ← 关键：入口播种，保证确定性
    
    T = len(measurements)
    
    # 初始化粒子：在真实初始状态附近采样
    particles = initial_state + np.random.randn(n_particles) * 2.0
    weights = np.ones(n_particles) / n_particles
    
    # 历史记录
    estimates = np.empty(T)
    particles_history = np.empty((T, n_particles))
    weights_history = np.empty((T, n_particles))
    
    for t in range(T):
        # ────────────────────────────────────
        # 1. 预测 (Prediction)
        # ────────────────────────────────────
        # 状态方程：x_t = x_{t-1} + v_t
        particles += np.random.randn(n_particles) * process_noise
        
        # ────────────────────────────────────
        # 2. 更新 (Update)
        # ────────────────────────────────────
        # 观测似然：p(z_t | x_t) = N(z_t; x_t, σ_measure²)
        z_t = measurements[t]
        for i in range(n_particles):
            # 高斯似然
            residual = z_t - particles[i]
            likelihood = np.exp(-0.5 * (residual / measurement_noise) ** 2)
            weights[i] *= likelihood
        
        # ────────────────────────────────────
        # 3. 归一化权重
        # ────────────────────────────────────
        weights_sum = np.sum(weights)
        if weights_sum > 1e-15:
            weights /= weights_sum
        else:
            # 所有粒子权重退化，重置为均匀
            weights = np.ones(n_particles) / n_particles
        
        # ────────────────────────────────────
        # 4. 状态估计（加权平均）
        # ────────────────────────────────────
        estimates[t] = np.sum(particles * weights)
        
        # 记录历史
        particles_history[t] = particles.copy()
        weights_history[t] = weights.copy()
        
        # ────────────────────────────────────
        # 5. 重采样（防止粒子退化）
        # ────────────────────────────────────
        N_eff = effective_sample_size(weights)
        if N_eff < n_particles / 2.0:
            indices = systematic_resample(weights)
            particles = particles[indices]
            weights = np.ones(n_particles) / n_particles
    
    return estimates, particles_history, weights_history
```

### 12.5.3 多 Seed 集成策略

```python
@njit
def multi_seed_particle_filter(
    measurements,
    n_particles,
    process_noise,
    measurement_noise,
    initial_state,
    base_seed,
    n_seeds           # 运行次数
):
    """
    多 Seed 似然加权粒子滤波
    
    步骤：
      1. 用 K 个不同 seed 运行 K 次粒子滤波
      2. 对每个 seed 的估计，计算其观测似然（作为置信度）
      3. 用似然加权平均 K 个估计
      4. 返回加权均值 + 标准差
    """
    T = len(measurements)
    
    # 存储 K 次运行的结果
    all_estimates = np.empty((n_seeds, T))
    all_likelihoods = np.empty(n_seeds)
    
    for k in range(n_seeds):
        seed_k = base_seed + k * 1000  # 确保 seed 之间差异足够大
        
        estimates_k, _, _ = particle_filter_single_seed(
            measurements, n_particles, process_noise,
            measurement_noise, initial_state, seed_k
        )
        
        all_estimates[k] = estimates_k
        
        # ────────────────────────────────────
        # 计算该估计序列的总似然（作为置信度）
        # ────────────────────────────────────
        log_likelihood = 0.0
        for t in range(T):
            residual = measurements[t] - estimates_k[t]
            log_likelihood += -0.5 * (residual / measurement_noise) ** 2
        
        all_likelihoods[k] = np.exp(log_likelihood)
    
    # ────────────────────────────────────
    # 似然归一化为权重
    # ────────────────────────────────────
    likelihood_sum = np.sum(all_likelihoods)
    if likelihood_sum > 1e-15:
        seed_weights = all_likelihoods / likelihood_sum
    else:
        # 全部似然很小，退化为等权重
        seed_weights = np.ones(n_seeds) / n_seeds
    
    # ────────────────────────────────────
    # 加权平均：每个时间步的估计
    # ────────────────────────────────────
    final_estimates = np.empty(T)
    final_std = np.empty(T)
    
    for t in range(T):
        # 加权均值
        final_estimates[t] = np.sum(seed_weights * all_estimates[:, t])
        
        # 加权标准差（量化不确定性）
        variance = np.sum(seed_weights * (all_estimates[:, t] - final_estimates[t]) ** 2)
        final_std[t] = np.sqrt(variance)
    
    return final_estimates, final_std, all_estimates, seed_weights
```

### 12.5.5 关键设计点解析

| 步骤 | 关键点 | 原因 |
|------|--------|------|
| **1. 入口播种** | `np.random.seed(seed)` 在函数第一行 | 确保每个 seed 的运行完全确定性（ch05 §5.6.5） |
| **2. Seed 偏移** | `seed_k = base_seed + k * 1000` | 确保不同 seed 之间充分独立 |
| **3. 似然加权** | 用观测似然作为 seed 权重 | "好"的估计（似然高）贡献更大 |
| **4. 系统重采样** | `systematic_resample()` | O(N) 复杂度 + 低方差 |
| **5. 有效粒子数** | 检查 `N_eff < N/2` 才重采样 | 避免过度重采样导致粒子多样性丧失 |
| **6. 权重归一化防护** | 检查 `weights_sum > 1e-15` | 防止数值下溢导致除零错误 |

## 12.7 本章关键经验

| 经验 | 说明 |
|------|------|
| **先 profile，再优化** | 用 cProfile 找热点，只优化占比高的部分 |
| **向量化 vs 循环** | 简单操作用 NumPy 向量化，复杂逻辑用 Numba 循环 |
| **并行的阈值** | 数据量 > 10万 + 计算密集 → 才考虑 `parallel=True` |
| **随机数要确定性** | 机器学习实验用 `np.random.seed()` 在函数入口（ch05 §5.6） |
| **预分配输出** | 重复调用的函数用 out parameter（ch06 §6.4） |
| **类型稳定** | 避免变量类型在循环中变化（ch11 §11.5.2） |

## 12.8 总结：何时用 Numba

```
✅ 应该用 Numba:
  - 循环密集、无法向量化
  - 自定义距离度量、相似度计算
  - 蒙特卡罗模拟、数值求解
  - 时序数据的滑动窗口操作
  - 需要并行加速的数组操作

❌ 不该用 Numba:
  - 纯 NumPy 向量化操作（np.sum(a+b)）
  - I/O 密集型操作
  - 需要 Python 对象（list/dict/str）
  - 代码只运行一次且很快（热身成本 > 收益）
```
